local Library = loadstring(game:HttpGet("https://raw.githubusercontent.com/nahida-cn/Roblox/main/UI-Library.lua"))()

local Window = Library:Window("光武", "极速传奇")

local Tab1 = Window:Tab("传送")

Tab1:Button("", function()
	Callback = function(Value)
		autoCrystal = Value
		while autoCrystal do
			game:GetService("ReplicatedStorage").rEvents.orbEvent:FireServer("collectOrb", "Gem", "City")
			wait()
		end
	end    
end)
